<script>

    // ⚠️ ВАЖНО: Замените на URL вашего Apps Script веб-приложения (который скопировали в Шаге 2)

    const GOOGLE_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbxZeRvOc6oT7-jZ7YjpxpoK3l1UWAEcvKluoy1qAgo9sE0lU-7604yx7-2Ky9GD8hKU7g';

    

    let html5QrCode = null;

    let isScanning = false;

    

    // Переключение вкладок

    function switchTab(tab) {

        document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));

        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

        document.getElementById(`${tab}-panel`).classList.add('active');

        event.target.classList.add('active');

        

        if (tab === 'catalog') loadCatalog();

        if (tab === 'scan') {

            // Если ушли со сканера - останавливаем

            if (html5QrCode && isScanning) {

                html5QrCode.stop().then(() => { isScanning = false; });

            }

        }

    }

    

    // Запуск сканера

    async function startScanner() {

        const resultDiv = document.getElementById('scan-result');

        resultDiv.innerHTML = '<div class="loading">📷 Запуск камеры...</div>';

        

        if (html5QrCode && isScanning) {

            await html5QrCode.stop();

        }

        

        html5QrCode = new Html5Qrcode("reader");

        

        const config = {

            fps: 10,

            qrbox: { width: 300, height: 300 },

            aspectRatio: 1.0

        };

        

        try {

            await html5QrCode.start(

                { facingMode: "environment" },

                config,

                (decodedText) => {

                    // Успешное сканирование

                    handleScanResult(decodedText);

                },

                (errorMessage) => {

                    console.log("Сканирование...", errorMessage);

                }

            );

            isScanning = true;

            resultDiv.innerHTML = '<div class="success">✅ Камера работает. Наведите на QR-код.</div>';

        } catch (err) {

            console.error('Ошибка:', err);

            resultDiv.innerHTML = '<div class="error">❌ Не удалось запустить камеру. Проверьте разрешения и используйте HTTPS.</div>';

        }

    }

    

    // Обработка результата сканирования

    async function handleScanResult(qrData) {

        // Останавливаем сканер после успешного сканирования

        if (html5QrCode && isScanning) {

            await html5QrCode.stop();

            isScanning = false;

        }

        

        const resultDiv = document.getElementById('scan-result');

        resultDiv.innerHTML = '<div class="loading">🔍 Поиск товара...</div>';

        

        try {

            const product = await findProductBySerial(qrData);

            displayProduct(product, 'scan-result');

        } catch (error) {

            resultDiv.innerHTML = `<div class="error">❌ ${error.message}</div>`;

        }

    }

    

    // Поиск по серийному номеру (через API)

    async function searchBySerial() {

        const serialNumber = document.getElementById('serial-input').value.trim();

        const resultDiv = document.getElementById('search-result');

        

        if (!serialNumber) {

            resultDiv.innerHTML = '<div class="error">❌ Введите серийный номер</div>';

            return;

        }

        

        resultDiv.innerHTML = '<div class="loading">🔍 Поиск...</div>';

        

        try {

            const product = await findProductBySerial(serialNumber);

            displayProduct(product, 'search-result');

        } catch (error) {

            resultDiv.innerHTML = `<div class="error">❌ ${error.message}</div>`;

        }

    }

    

    // Запрос к Google Apps Script API

    async function findProductBySerial(serialNumber) {

        const url = `${GOOGLE_SCRIPT_URL}?serial=${encodeURIComponent(serialNumber)}`;

        const response = await fetch(url);

        const data = await response.json();

        

        if (data.error) {

            throw new Error(data.error);

        }

        return data;

    }

    

    // Отображение товара

    function displayProduct(product, containerId) {

        const container = document.getElementById(containerId);

        

        let затраты = product['Затраты'];

        if (затраты && !isNaN(parseFloat(затраты))) {

            затраты = `${parseFloat(затраты).toLocaleString('ru-RU')} ₽`;

        }

        

        let html = `

            <div class="product-card">

                <h3>🚁 ${escapeHtml(product['Название'] || 'Без названия')}</h3>

                <div class="product-detail">

                    <strong>📇 Серийный номер:</strong> ${escapeHtml(product['Серийный номер'] || '—')}

                </div>

                <div class="product-detail">

                    <strong>📝 Описание:</strong> ${escapeHtml(product['Описание'] || '—')}

                </div>

                <div class="product-detail">

                    <strong>💰 Затраты:</strong> ${затраты || '—'}

                </div>

                <div class="product-detail">

                    <strong>🛒 Закупка:</strong> ${escapeHtml(product['Закупка'] || '—')}

                </div>

        `;

        

        if (product['Ссылка на объявление Авито'] && product['Ссылка на объявление Авито'] !== '') {

            html += `

                <div class="product-detail">

                    <strong>🔗 Авито:</strong> 

                    <a href="${escapeHtml(product['Ссылка на объявление Авито'])}" target="_blank">Открыть объявление →</a>

                </div>

            `;

        } else {

            html += `<div class="product-detail"><strong>🔗 Авито:</strong> —</div>`;

        }

        

        if (product['QR_Code'] && product['QR_Code'] !== '') {

            html += `

                <div class="qr-code">

                    <strong>📱 QR-код:</strong><br>

                    <img src="${escapeHtml(product['QR_Code'])}" style="max-width: 120px;">

                </div>

            `;

        }

        

        html += `</div>`;

        container.innerHTML = html;

    }

    

    function escapeHtml(str) {

        if (str === undefined || str === null) return '';

        return String(str)

            .replace(/&/g, '&amp;')

            .replace(/</g, '&lt;')

            .replace(/>/g, '&gt;')

            .replace(/"/g, '&quot;')

            .replace(/'/g, '&#39;');

    }

    

    // Загрузка каталога

    async function loadCatalog() {

        const container = document.getElementById('catalog-content');

        container.innerHTML = '<div class="loading">📦 Загрузка...</div>';

        

        try {

            const products = await getAllProducts();

            if (products.length === 0) {

                container.innerHTML = '<div class="error">📭 Товары не найдены.</div>';

                return;

            }

            

            let html = '<div class="products-grid">';

            products.forEach(product => {

                html += `

                    <div class="product-item" onclick="viewProductDetail('${escapeHtml(product['Серийный номер'])}')">

                        <strong>${escapeHtml(product['Название'] || 'Без названия')}</strong><br>

                        <small>📇 ${escapeHtml(product['Серийный номер'] || '—')}</small>

                    </div>

                `;

            });

            html += '</div>';

            container.innerHTML = html;

        } catch (error) {

            container.innerHTML = `<div class="error">❌ ${error.message}</div>`;

        }

    }

    

    async function viewProductDetail(serialNumber) {

        document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));

        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

        document.getElementById('search-panel').classList.add('active');

        document.querySelector('.tab-btn:nth-child(2)').classList.add('active');

        document.getElementById('serial-input').value = serialNumber;

        await searchBySerial();

    }

    

    async function getAllProducts() {

        // Получаем все товары через отдельный API-запрос

        const url = `${GOOGLE_SCRIPT_URL}?action=getAll`;

        const response = await fetch(url);

        return await response.json();

    }

    

    document.getElementById('start-scan-btn').onclick = startScanner;

    

    window.onload = function() {

        loadCatalog();

    };
